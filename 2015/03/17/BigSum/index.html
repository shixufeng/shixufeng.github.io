大整数求和 

#include <iostream>
#include <vector>
#include <string>
#define N 1000
using namespace std;

int main()
{
    int T;
    cin>>T;
    vector<string> svec;
    for(int i=1;i!=T+1;i++)
    {
        string A,B;
        cin>>A>>B;
        svec.push_back(A);
        svec.push_back(B);
    }

    int count=1;  // case count:

    for(vector<string>::iterator iter=svec.begin();iter!=svec.end();iter=iter+2)
    {    
        int sum[N];  // save summation

        int LenA=(*iter).length(),LenB=(*(iter+1)).length();
        int i=LenA-1,j=LenB-1,k=0;
        int m=0,n=0,temp=0;

        // 开始求和，结果倒序
        while(i>=0 && j>=0)
        {
            m=(*iter)[i]-'0';
            n=(*(iter+1))[j]-'0';

            sum[k]=(m+n+temp)%10;
            temp=(m+n+temp)/10;

            i--;
            j--;
            k++;
        }
        if(i>j)
        {
            while(i>=0)
            {
                m=(*iter)[i]-'0';
                sum[k]=(m+temp)%10;
                temp=(m+temp)/10;
                k++;
                i--;
            }
        }
        if(i<j)
        {
            while(j>=0)
            {
                n=(*(iter+1))[j]-'0';
                sum[k]=(n+temp)%10;
                temp=(n+temp)/10;
                k++;
                j--;
            }
        }
        if(temp!=0)
        {
            sum[k]=temp;
            k++;
        }

        cout<<"Case"<<" "<<count<<":"<<endl;
        cout<<*iter<<" "<<"+"<<" "<<*(iter+1)<<" "<<"="<<" ";
        for(int index1=k-1;index1>=0;index1--)
        {
            cout<<sum[index1];
        }
        if(count>=1 && count<T)
            cout<<endl<<endl;
        else
            cout<<endl;
        count++;
    }
    return 0;
}
